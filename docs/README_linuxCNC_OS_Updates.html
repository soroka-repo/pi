# Compile the latest level of Blenter for the PI
 sudo apt update
 sudo apt install git build-essential cmake libx11-dev libxxf86vm-dev libxcursor-dev libxi-dev libxrandr-dev libxinerama-dev libglew-dev libopenexr-dev libopenimageio-dev libpython3-dev 
 sudo apt install libavdevice-dev libavformat-dev libavcodec-dev libswscale-dev libavutil-dev
 sudo apt install autoconf automake bison libtool yasm tcl ninja-build meson python3-mako patchelf pkg-config
 sudo apt install git-lfs 
 
 mkdir -p ~/github
 cd ~/github
 GIT_LFS_SKIP_SMUDGE=1;
 git clone --recursive https://github.com/blender/blender.git ; 
 cd blender;
 git checkout -b v4.5.0; # or the latest stable version
 git submodule update --init --recursive;
 ./build_files/build_environment/install_linux_packages.py --all;
 make deps
 make update
 #git lfs install
 #git lfs fetch --all; ### git lfs pull
 #git lfs checkout;
 
 
 # go to https://download.blender.org/source/ and download the source file and untar the file
  tar -xf blender-4.5.0.tar.xz  
  cd blender-4.5.0
  ./build_files/build_environment/install_linux_packages.py --distro-id debian
  
 mkdir build_pi5;
 cd build_pi5;
 cmake .. \
    -DCMAKE_BUILD_TYPE=Release \
    -DWITH_CYCLES=OFF \
    -DWITH_OPENVDB=OFF \
    -DWITH_MOD_OCEAN=OFF \
    -DWITH_AUDASPACE=OFF \
    -DWITH_BLENDER=ON \
    -DWITH_PYTHON_INSTALL=OFF \
    -DWITH_PYTHON_MODULE=OFF \
    -DWITH_INSTALL_PORTABLE=ON \
    -DCMAKE_CXX_FLAGS="-march=armv8-a+crc+crypto -O3" \
    -DCMAKE_C_FLAGS="-march=armv8-a+crc+crypto -O3" \
    -DWITH_VULKAN_BACKEND=ON;
### Key Flag Breakdown for Performance:
#    -DCMAKE_BUILD_TYPE=Release: Disables debug symbols, increasing speed.
#    -DWITH_CYCLES=OFF: Drastically speeds up compilation and reduces memory footprint if you are only modeling.
#    -march=armv8-a+crc+crypto -O3: Tells GCC to optimize for ARMv8 with CRC and Crypto extensions (supported by Pi 5) and enables aggressive optimization.
#    -DWITH_VULKAN_BACKEND=ON: Essential for utilizing the Pi 5's improved GPU drivers (via Igalia/Mesa) for viewport modeling.
#    -DWITH_AUDASPACE=OFF: Removes audio dependencies, unnecessary for modeling. 
 make -j$(nproc) install

    STEP: Mount an extra NVME disk (just using the disk to listen to music with VLC while I configure the PI and write this doc...)
    
###################################
 lsblk ### to look for your new disk
 sudo mkdir -p /mnt/nvme  ### such that "nvme" is the name of your new disk
 sudo mount /dev/nvme0n1p2 /mnt/nvme ### test the mount, this will mount the disk for this boot session only
                                ### note: all disks start with /dev and nvme0n1p2 is what you found from lsblk
### Make the disk you mounted above presistent across boots
 lsblk -f ### and find the "UUID" of the disk also note the "FSTYPE" which i ext4 in this example 
 sudo vi /etc/fstab ### edit the fstab file and add the following at the bottom to make the mount presistent 
   
UUID=f6bb7216-cc28-4ec3-86d6-b391f5f31405	/mnt/nvme	ext4 	defaults

    STEP: Networking adding an extra interface for LinuxCNC
    
###################################
### add an extra USB to 1g Ethernet converter to the blue USB connector. blue = USB-3 so faster then the USB-2 to the right
 lsusb ### will check that the USB Ethernet is reconized by the PI
 lsusb -v ### for a more verbose listing
 ip a ### lists all the network interfaces, mac addresses, ip addresses... eth0 is the onboard, eth1 will be the new interface
      ### wlan is the wifi interface
 ### The default IP address for the Mesa 7i96 is either 10.10.10.10 or 192.168.1.121 depending on it's jumper configurations
 ### I'm choosing the jumpers to set the Mesa to 10.10.10.10 so I'm going to set my extra (eth1) IP address to 10.10.10.1  
 ### So edit the file "/etc/network/interfaces" and add the following after the eth0 setction:


    EXAMPLE:

    
auto eth1
iface eth1 inet static
    address 10.10.10.1
    netmask 255.255.255.0

    

 sudo su
 vi /etc/network/interfaces ### to edit and add the above into the file
 systemctl restart networking ### to restart the network


 ip a ### will show that eth1 has been set to 10.10.10.1
ping 10.10.10.10 ### connect the ethernet cable to the Mesa 7i96 and power it on and the ping will respond <CTRL>C to stop the ping

    EXAMPLE:

    
 ping 10.10.10.10
